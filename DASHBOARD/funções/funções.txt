calculateHours: function(hour){

      let [hora, minutos, segundos] = hour.split(':').map(element => parseInt(element));
      let convertMinutes = 0
      let convertSeconds = 0
      let hourConvert = 0

      if(minutos > 0 && segundos > 0){

        convertMinutes = minutos / 60;
        convertSeconds = segundos / 60;
        
        hourConvert = hora + convertMinutes + convertSeconds;

      } else if (minutos > 0){

        convertMinutes = minutos / 60;

        hourConvert = hora + convertMinutes + segundos;


      } else if (segundos > 0){

        convertSeconds = segundos / 60;

        hourConvert = hora + minutos + convertSeconds;

      } else {

        hourConvert = hora + minutos + segundos;

      }

      return hourConvert;


}
